### Task 3: Extract and enhance check-docs validation

**Files:**
- Create: `scripts/check-docs.sh`
- Modify: `Makefile` (check-docs target calls script)

**Context:** The current `check-docs` target is inline bash in the Makefile (~20 lines). Extract it to a script and add 4 new validation checks from the spec.

- [ ] **Step 1: Create scripts/check-docs.sh**

```bash
#!/usr/bin/env bash
set -euo pipefail

# check-docs.sh — CI gate: validate documentation consistency
#
# Checks:
#   1. Agent references — every agent in .claude/agents/ appears in CLAUDE.md, README.md,
#      docs/architecture.md, and .claude/rules/use-pipeline-agents.md
#   2. Skill references — every user-facing skill appears in CLAUDE.md
#   3. Shim function sync — transformation-rules.md ↔ header files (bidirectional)
#   4. Duplication regression — CLAUDE.md must not contain content from canonical rules
#
# Exit 0 if all pass, exit 1 if any fail.

FAIL=0

# ============================================================
# Check 1: Agent cross-reference
# ============================================================
echo "=== Check 1: Agent references ==="
AGENTS=$(ls .claude/agents/*.md 2>/dev/null | xargs -I{} basename {} .md | sort)
agent_fail=0

for agent in $AGENTS; do
    for doc in CLAUDE.md README.md docs/architecture.md; do
        if ! grep -q "$agent" "$doc" 2>/dev/null; then
            echo "FAIL  agent '$agent' not found in $doc"
            agent_fail=1
        fi
    done
    # Also check use-pipeline-agents.md
    if ! grep -q "$agent" .claude/rules/use-pipeline-agents.md 2>/dev/null; then
        echo "FAIL  agent '$agent' not found in .claude/rules/use-pipeline-agents.md"
        agent_fail=1
    fi
done

# Reverse: agents referenced in use-pipeline-agents.md must exist
for agent_ref in $(grep -oE '`[a-z_-]+` agent' .claude/rules/use-pipeline-agents.md 2>/dev/null | sed 's/`//g; s/ agent//' | sort -u); do
    if [ ! -f ".claude/agents/${agent_ref}.md" ]; then
        echo "FAIL  agent '$agent_ref' in use-pipeline-agents.md has no .claude/agents/${agent_ref}.md"
        agent_fail=1
    fi
done

if [ "$agent_fail" -eq 0 ]; then
    echo "PASS  All $(echo "$AGENTS" | wc -w | tr -d ' ') agents referenced in all docs"
fi
FAIL=$((FAIL + agent_fail))

# ============================================================
# Check 2: Skill references
# ============================================================
echo ""
echo "=== Check 2: Skill references ==="
skill_fail=0

for skill_dir in .claude/skills/*/; do
    skill_name=$(basename "$skill_dir")
    skill_file="$skill_dir/SKILL.md"
    [ -f "$skill_file" ] || continue

    # Check if this is a user-facing skill (has argument-hint or is a top-level command)
    # Skills that are sub-skills (called by other skills) don't need CLAUDE.md references
    if grep -q '^argument-hint:' "$skill_file" 2>/dev/null; then
        # Has argument-hint = user-facing slash command
        if ! grep -q "$skill_name" CLAUDE.md 2>/dev/null; then
            echo "FAIL  skill '/$skill_name' (user-facing) not found in CLAUDE.md"
            skill_fail=1
        fi
    fi
done

if [ "$skill_fail" -eq 0 ]; then
    skill_count=$(find .claude/skills/ -name 'SKILL.md' -maxdepth 2 | wc -l | tr -d ' ')
    echo "PASS  All user-facing skills referenced in CLAUDE.md ($skill_count total skills)"
fi
FAIL=$((FAIL + skill_fail))

# ============================================================
# Check 3: Shim function sync (bidirectional)
# ============================================================
echo ""
echo "=== Check 3: Shim function sync ==="
shim_fail=0
TRANSFORM_RULES=".claude/skills/transform-source/references/transformation-rules.md"

if [ -f "$TRANSFORM_RULES" ]; then
    # Direction A: functions in transformation-rules.md must exist in headers
    rules_funcs=$(grep -oE 'amiport_[a-z_]+' "$TRANSFORM_RULES" 2>/dev/null | sort -u)
    for func in $rules_funcs; do
        # Skip emu functions (different library)
        case "$func" in amiport_emu_*) continue ;; esac
        # Check if function exists in any shim header
        if ! grep -rq "$func" lib/posix-shim/include/amiport/ 2>/dev/null; then
            echo "FAIL  '$func' in transformation-rules.md but not in shim headers (stale?)"
            shim_fail=1
        fi
    done

    # Direction B: POSIX-replacement functions in headers should be in transformation-rules.md
    # Exclusion list: internal helpers with no POSIX equivalent
    HELPERS="amiport_errno_from_ioerr amiport_expand_argv amiport_check_break amiport_fd_is_valid amiport_fd_to_fh amiport_exit amiport_free_argv amiport_map_errno amiport_glob_has_magic amiport_file_count amiport_file_entry amiport_files amiport_dirent amiport_optarg amiport_opterr amiport_optind amiport_optopt amiport_optreset amiport_glob_t"
    header_funcs=$(grep -rohE 'amiport_[a-z_]+' lib/posix-shim/include/amiport/*.h 2>/dev/null | sort -u)
    for func in $header_funcs; do
        # Skip helpers
        skip=false
        for helper in $HELPERS; do
            [ "$func" = "$helper" ] && skip=true && break
        done
        [ "$skip" = true ] && continue

        if ! grep -q "$func" "$TRANSFORM_RULES" 2>/dev/null; then
            echo "WARN  '$func' in shim headers but not in transformation-rules.md (new function?)"
            # WARN not FAIL — new functions are expected during development
        fi
    done

    if [ "$shim_fail" -eq 0 ]; then
        echo "PASS  Transformation rules consistent with shim headers"
    fi
else
    echo "SKIP  $TRANSFORM_RULES not found"
fi
FAIL=$((FAIL + shim_fail))

# ============================================================
# Check 4: Duplication regression guard
# ============================================================
echo ""
echo "=== Check 4: Duplication regression ==="
dup_fail=0

# Marker strings from canonical rules that must NOT appear in CLAUDE.md
# These indicate content has crept back from rules files
MARKERS=(
    "fopen() macro collision"
    "vamos argv pointer"
    "vsnprintf(NULL"
    "pledge/unveil stubs"
    "### Epoch offset"
    "### __nowild for"
    "### vamos ignores __stack"
    "### st_dev/st_ino"
    "ANSI C89"
)

for marker in "${MARKERS[@]}"; do
    if grep -q "$marker" CLAUDE.md 2>/dev/null; then
        echo "FAIL  CLAUDE.md contains '$marker' — this belongs in .claude/rules/ not CLAUDE.md"
        dup_fail=1
    fi
done

if [ "$dup_fail" -eq 0 ]; then
    echo "PASS  No duplicated content detected in CLAUDE.md"
fi
FAIL=$((FAIL + dup_fail))

# ============================================================
# Summary
# ============================================================
echo ""
if [ "$FAIL" -gt 0 ]; then
    echo "FAILED: $FAIL check(s) with errors"
    exit 1
else
    echo "All documentation consistency checks passed."
fi
```
