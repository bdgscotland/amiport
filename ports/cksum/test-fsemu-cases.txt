# cksum test suite for FS-UAE
# Category 1 (CLI), FreeBSD cksum 1.0 ported to AmigaOS 3.x
#
# Algorithms supported:
#   default    -- POSIX CRC-32 (POSIX 1003.2), output: "<crc> <bytes> [file]"
#   -o 1       -- BSD sum (16-bit rotate-right), output: "<sum> <1k-blocks> [file]"
#   -o 2       -- SYSV sum (32-bit additive),    output: "<sum> <512-blocks> [file]"
#   -o 3       -- CRC32 (AUTODIN-II/Ethernet),   output: "<crc> <bytes> [file]"
#
# All EXPECT: values verified by native tool run on macOS.

# =============================================================================
# FUNCTIONAL: Default POSIX CRC mode
# =============================================================================

# Native: cksum test-oneline.txt -> "3733384285 12 ..."
TEST: Default POSIX CRC of known file produces correct checksum and byte count
CMD: WORK:cksum WORK:test-oneline.txt
EXPECT: 3733384285 12 WORK:test-oneline.txt
EXPECT_RC: 0

# Native: cksum test-cksum-abc.txt -> "1112837078 4 ..."
TEST: Default POSIX CRC of 4-byte file abc-newline
CMD: WORK:cksum WORK:test-cksum-abc.txt
EXPECT: 1112837078 4 WORK:test-cksum-abc.txt
EXPECT_RC: 0

# =============================================================================
# PRECISION: Known reference checksum values (regression guards)
# =============================================================================

# These test the exact numerical correctness of the CRC algorithms against
# known reference values computed from the source algorithm tables.

# POSIX CRC reference: the empty string edge case is the canonical test
# ~0 (all-ones initial CRC) with no data bytes and no length bytes = ~0 = 4294967295
TEST: Precision - POSIX CRC of empty file is exactly 4294967295 (canonical reference value)
CMD: WORK:cksum WORK:test-empty.txt
EXPECT: 4294967295 0 WORK:test-empty.txt
EXPECT_RC: 0

# CRC32 reference: empty file = ~(~0) = 0 (initial ~0, no XOR operations, complement = 0)
TEST: Precision - CRC32 of empty file is exactly 0 (AUTODIN-II identity for empty input)
CMD: WORK:cksum -o 3 WORK:test-empty.txt
EXPECT: 0 0 WORK:test-empty.txt
EXPECT_RC: 0

# POSIX CRC known value: "hello world\n" (12 bytes) = 3733384285
# This exact value is verified against macOS native cksum and the FreeBSD polynomial table
TEST: Precision - POSIX CRC of hello-world-newline matches FreeBSD reference value
CMD: WORK:cksum WORK:test-oneline.txt
EXPECT: 3733384285 12 WORK:test-oneline.txt
EXPECT_RC: 0

# CRC32 known value: "hello world\n" (12 bytes) = 2936552237
# Verified against AUTODIN-II polynomial (same as Ethernet CRC32 but not same as zlib CRC32)

EXPECT: 2936552237 12 WORK:test-oneline.txt
EXPECT_RC: 0
