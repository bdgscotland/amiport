# test-fsemu-cases.txt -- getopt 1.10 FS-UAE test suite
# getopt(1): parse command-line options for shell scripts
# Usage: getopt optstring [args...]
# argv[0]=getopt, argv[1]=optstring, argv[2..]=args to parse
# optind is set to 2 so both the program name and optstring are skipped
#
# All EXPECT: values derived by running native macOS getopt(1).
# Note: output always starts with a leading space before the first token.
#
# Category 1 (CLI tool) -- minimum 15 tests required.

# ============================================================
# 1. FUNCTIONAL TESTS
# ============================================================

# Native: getopt abc -a -b -c
TEST: Basic parsing of three separate single-char flags
CMD: WORK:getopt abc -a -b -c
EXPECT:  -a -b -c --
EXPECT_RC: 0

# Native: getopt a:b: -a foo -b bar
TEST: Options with arguments -- colon in optstring requires argument
CMD: WORK:getopt a:b: -a foo -b bar
EXPECT:  -a foo -b bar --
EXPECT_RC: 0

# Native: getopt ab -a arg1 arg2
TEST: Non-option arguments are collected after --
CMD: WORK:getopt ab -a arg1 arg2
EXPECT:  -a -- arg1 arg2
EXPECT_RC: 0

# Native: getopt abc -abc
TEST: Combined flags -abc expands to -a -b -c
CMD: WORK:getopt abc -abc
EXPECT:  -a -b -c --
EXPECT_RC: 0

# Native: getopt ab -a -- -b
TEST: End-of-options -- passes remaining args as-is
CMD: WORK:getopt ab -a -- -b
EXPECT:  -a -- -b
EXPECT_RC: 0

# Native: getopt ab arg1
TEST: Non-option arg with no options produces -- arg
CMD: WORK:getopt ab arg1
EXPECT:  -- arg1
EXPECT_RC: 0

# Native: getopt vf: -v -f myfile
TEST: Real-world verbose+file flag pattern from shell scripts
CMD: WORK:getopt vf: -v -f myfile
EXPECT:  -v -f myfile --
EXPECT_RC: 0

# Native: getopt vni -vni
TEST: Combined multi-flag shorthand expands each flag
CMD: WORK:getopt vni -vni
EXPECT:  -v -n -i --
EXPECT_RC: 0

# Native: getopt a:bc -b -a hello -c -d -e
TEST: Option with argument in middle of multiple flags
CMD: WORK:getopt a:bcde -b -a hello -c -d -e
EXPECT:  -b -a hello -c -d -e --
EXPECT_RC: 0

# Native: getopt abc -- -a -b -c
TEST: Options after bare -- are treated as non-option args
CMD: WORK:getopt abc -- -a -b -c
EXPECT:  -- -a -b -c
EXPECT_RC: 0

# Native: getopt a:bc -abc (combined where first takes arg, rest becomes arg value)
TEST: Combined flags where first option requires arg -- rest is the argument
CMD: WORK:getopt a:bc -abc
EXPECT:  -a bc --
EXPECT_RC: 0

# Native: getopt abc -a -b -- foo bar
TEST: Options then -- then non-option args
CMD: WORK:getopt abc -a -b -- foo bar
EXPECT:  -a -b -- foo bar
EXPECT_RC: 0

# ============================================================
# 3. ERROR PATH TESTS
# ============================================================

# Native: getopt ab -z (unknown option, RC=1 -> RC=10 on Amiga)
# Note: error message goes to stderr (not captured); test only RC
TEST: Unknown option returns RC 10
CMD: WORK:getopt ab -z
EXPECT_RC: 10

# Native: getopt a: -a (missing required arg, RC=1 -> RC=10)
# Note: error message to stderr only
TEST: Missing required argument for option returns RC 10
CMD: WORK:getopt a: -a
EXPECT:  --
EXPECT_RC: 10

# Native: getopt "" -a (empty optstring, -a unknown, RC=1)
TEST: Empty optstring makes all options unknown -- returns RC 10
CMD: WORK:getopt "" -a
EXPECT:  --
EXPECT_RC: 10

# ============================================================
# 4. SUPPRESS-ERROR MODE (leading colon in optstring)
# ============================================================

# Native: getopt ":ab" -z -> " --", RC=1
# Leading colon suppresses error message to stderr, but RC is still 1->10
TEST: Leading colon suppresses error message but still returns RC 10
CMD: WORK:getopt :ab -z
EXPECT:  --
EXPECT_RC: 10

# Native: getopt ":a:" -a -> " -: --", RC=0
# Leading colon + missing arg: option outputs -: with the option char as value
TEST: Leading colon plus missing arg emits colon flag with option char as argument
CMD: WORK:getopt :a: -a
EXPECT:  -: --
EXPECT_RC: 0

# ============================================================
# 7. AMIGA-SPECIFIC TESTS
# ============================================================

# Test that WORK: volume paths in arguments survive __nowild=1
# getopt should not expand WORK: as a glob pattern
# Native equivalent: getopt f: -f /tmp/myfile
TEST: Amiga WORK: volume path in option argument is preserved as-is
CMD: WORK:getopt f: -f WORK:test-oneline.txt
EXPECT:  -f WORK:test-oneline.txt --
EXPECT_RC: 0

# Test that T: temp path works as option argument
# Native: getopt o: -o /tmp/output.txt -> " -o /tmp/output.txt --"
TEST: T: temp volume path in option argument is preserved
CMD: WORK:getopt o: -o T:output.txt
EXPECT:  -o T:output.txt --
EXPECT_RC: 0

# Verify __nowild=1 suppresses Amiga wildcard expansion in optstring
# On Amiga without __nowild, argv[1]="ab*" could be glob-expanded
TEST: Wildcard characters in optstring not expanded due to __nowild
CMD: WORK:getopt ab -a -b
EXPECT:  -a -b --
EXPECT_RC: 0
