# test-fsemu-cases.txt -- rs (reshape data array) test suite
# Category 1 (CLI tool) -- minimum 15 tests required
# rs reads from stdin only -- all CMD lines use AmigaDOS < redirection.
# Since CMD is written into an Execute script, AmigaDOS shell parses < as
# stdin redirection. Pattern: "CMD: WORK:rs [args] <WORK:input.txt"
# See ports/colrm/test-fsemu-cases.txt for the established precedent.
# getopt option string: "c::C::s::S::k:K:g:G:w:tTeEnyjhHmz"

# ===========================================================================
# FUNCTIONAL TESTS -- one test per flag
# ===========================================================================

# Native: rs 2 3 < test-rs-6elem.txt
TEST: Basic reshape 6 elements into 2 rows x 3 cols
CMD: WORK:rs 2 3 <WORK:test-rs-6elem.txt
EXPECT: a  b  c
EXPECT_LINE: 2,d  e  f
EXPECT_RC: 0

# Native: rs -t < test-rs-2x3.txt
TEST: Transpose -t flattens 2x3 matrix to single row
CMD: WORK:rs -t <WORK:test-rs-2x3.txt
EXPECT: a  b  c  d  e  f
EXPECT_RC: 0

# Native: rs -T < test-rs-2x3.txt
TEST: Matrix transpose -T swaps rows and cols (2x3 becomes 3x2)
CMD: WORK:rs -T <WORK:test-rs-2x3.txt
EXPECT: a  d
EXPECT_LINE: 2,b  e
EXPECT_LINE: 3,c  f
EXPECT_RC: 0

# Native: rs -c: < test-rs-colon.txt
TEST: Input column separator -c: reads colon-delimited input strictly
CMD: WORK:rs -c: <WORK:test-rs-colon.txt
EXPECT: a  b  c  d  e  f
EXPECT_RC: 0

# Native: rs -C, < test-rs-2x3.txt
TEST: Output column separator -C, writes comma only between columns
CMD: WORK:rs -C, <WORK:test-rs-2x3.txt
EXPECT: a,b,c,d,e,f,
EXPECT_RC: 0

# Native: rs -S, < test-rs-2x3.txt

EXPECT_RC: 0

# Native: rs -h < test-rs-2x3.txt

EXPECT: 2 3
EXPECT_RC: 0

# Native: rs -H < test-rs-2x3.txt

EXPECT_RC: 0
