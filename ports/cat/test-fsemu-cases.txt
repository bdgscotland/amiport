# test-fsemu-cases.txt -- FS-UAE test suite for cat 1.34
# Category 1 (CLI utility). Minimum: 8 tests.
# Flags: -b -e -n -s -t -u -v (getopt "benstuv")
# All EXPECT: values pre-computed by running native cat on macOS.

# =============================================================
# FUNCTIONAL TESTS -- one per flag
# =============================================================

# Native: cat test-oneline.txt | head -1
TEST: Basic cat of a file prints its content
CMD: WORK:cat WORK:test-oneline.txt
EXPECT: hello world
EXPECT_RC: 0

# Native: cat -n test-oneline.txt | head -1
TEST: -n numbers all lines with leading spaces and tab
CMD: WORK:cat -n WORK:test-oneline.txt
EXPECT:      1	hello world
EXPECT_RC: 0

# Native: cat -b test-cat-blank.txt | head -1
TEST: -b numbers only non-blank lines
CMD: WORK:cat -b WORK:test-cat-blank.txt
EXPECT:      1	first line
EXPECT_RC: 0

# Native: cat -e test-multiline.txt | head -1
# Note: test-oneline.txt has no trailing newline so -e produces no $; use multiline.txt
TEST: -e appends dollar sign at end of each line
CMD: WORK:cat -e WORK:test-multiline.txt
EXPECT: hello world$
EXPECT_RC: 0

# Native: cat -s test-cat-blank.txt | head -1
TEST: -s squeezes consecutive blank lines to one
CMD: WORK:cat -s WORK:test-cat-blank.txt
EXPECT: first line
EXPECT_RC: 0

# Native: cat -t test-cat-tabs.txt | head -1
TEST: -t displays tab characters as ^I
CMD: WORK:cat -t WORK:test-cat-tabs.txt
EXPECT: col1^Icol2^Icol3
EXPECT_RC: 0

# =============================================================
# ERROR PATH TESTS
# =============================================================

# cat source: returns 10 for bad option (fprintf(stderr, usage) + return 10)
TEST: Unknown flag prints usage and exits with RC 10
CMD: WORK:cat -Z WORK:test-oneline.txt
EXPECT_RC: 10

# cat source: warn() + rval=1 on missing file (raw_cat path returns rval which is 1)
# RC=1 on Amiga is non-zero but not standard AmigaDOS codes.
# The port sets rval=1 and returns it -- AmigaDOS sees RC=1 (equivalent to RETURN_WARN=5? no: 1)
# Source confirms: rval = 1; ... return rval. On Amiga RC=1 is treated as non-fatal.
# Use EXPECT_RC: 1 since that's what rval=1 returns.
TEST: Missing file prints warning and exits with non-zero RC
CMD: WORK:cat WORK:test-cat-nonexistent.txt
EXPECT_RC: 1
