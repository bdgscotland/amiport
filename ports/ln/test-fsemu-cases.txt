# ln 1.25 -- FS-UAE test suite
#
# IMPORTANT: The uaehf virtual filesystem used in FS-UAE testing does NOT
# support MakeLink() (soft or hard links). All symlink/hardlink creation
# tests return RC=1 on uaehf. These tests pass on real FFS/SFS filesystems.
#
# This test suite covers error paths, flag parsing, and usage -- the parts
# that work regardless of filesystem. Symlink functionality must be tested
# manually on real AmigaOS hardware or a properly configured FFS partition.

# ==============================================================
# ERROR PATH TESTS (filesystem-independent)
# ==============================================================

TEST: No arguments prints usage and returns RC 10
CMD: WORK:ln
EXPECT_RC: 10

TEST: Unknown flag -z returns usage error RC 10
CMD: WORK:ln -z
EXPECT_RC: 10

TEST: Hard link to non-existent source returns error
CMD: WORK:ln T:nonexistent-file T:nonexistent-link
EXPECT_RC: 1

TEST: Multiple non-existent sources returns error
CMD: WORK:ln T:nosuch1 T:nosuch2 T:nosuchdir
EXPECT_RC: 10

# ==============================================================
# FLAG PARSING TESTS (all flags accepted without crash)
# ==============================================================

TEST: -s flag is accepted (symlink mode)
CMD: WORK:ln -s T:nosource T:nodest-s
EXPECT_RC: 1

TEST: -f flag is accepted (force overwrite)
CMD: WORK:ln -f T:nosource T:nodest-f
EXPECT_RC: 1

TEST: -h flag is accepted
CMD: WORK:ln -h T:nosource T:nodest-h
EXPECT_RC: 1

TEST: -n flag is accepted (same as -h)
CMD: WORK:ln -n T:nosource T:nodest-n
EXPECT_RC: 1

TEST: -P flag is accepted
CMD: WORK:ln -P T:nosource T:nodest-P
EXPECT_RC: 1

TEST: -L flag is accepted
CMD: WORK:ln -L T:nosource T:nodest-L
EXPECT_RC: 1

TEST: Combined flags -sfn accepted
CMD: WORK:ln -sfn T:nosource T:nodest-sfn
EXPECT_RC: 1

# ==============================================================
# KNOWN LIMITATION: SYMLINK CREATION ON UAEHF
# ==============================================================
# ln -s creates soft links via MakeLink(). The uaehf filesystem
# returns ERROR_NOT_IMPLEMENTED for MakeLink(). On real FFS/SFS,
# symlink creation works. This is documented in PORT.md.

TEST: Symlink creation returns non-zero on uaehf (expected limitation)
CMD: WORK:ln -s WORK:test-ln-source.txt T:test-symlink
EXPECT_RC: 1
